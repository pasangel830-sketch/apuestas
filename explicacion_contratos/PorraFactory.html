PorraFactory.sol

Fábrica que crea porras: despliega WhitelistManager + PorraGame y los registra.

Código completo del contrato

 1// SPDX-License-Identifier: MIT
 2pragma solidity ^0.8.19;

 4import "./PorraGame.sol";
 5import "./WhitelistManager.sol";
 6import "./utils/Ownable.sol";

 8/// @title Factory para crear porras
 9/// @notice Despliega WhitelistManager + PorraGame y registra la porra
10contract PorraFactory is Ownable {
11    address public oracle;
12    PorraGame[] public porras;
13    mapping(address => address[]) public porrasByCreator;

15    event PorraCreated(
16        address indexed game,
17        address indexed whitelistManager,
18        address indexed creator,
19        bytes32 matchId,
20        uint256 bettingDeadline
21    );

23    constructor(address _oracle) Ownable(msg.sender) {
24        require(_oracle != address(0), "PorraFactory: zero oracle");
25        oracle = _oracle;
26    }

28    function createPorra(
29        address[] calldata participants,
30        bytes32 matchId,
31        uint256 bettingDeadline,
32        uint256 matchEndTime,
33        uint256 stakeAmount
34    ) external returns (address game, address whitelistManagerAddr) {
35        WhitelistManager wl = new WhitelistManager(participants);
36        whitelistManagerAddr = address(wl);

38        PorraGame g = new PorraGame(
39            whitelistManagerAddr,
40            oracle,
41            matchId,
42            bettingDeadline,
43            matchEndTime,
44            stakeAmount
45        );
46        game = address(g);

48        porras.push(g);
49        porrasByCreator[msg.sender].push(game);

51        emit PorraCreated(game, whitelistManagerAddr, msg.sender, matchId, bettingDeadline);
52        return (game, whitelistManagerAddr);
53    }

55    function getPorrasCount() external view returns (uint256) {
56        return porras.length;
57    }

59    function getPorraAt(uint256 index) external view returns (address) {
60        require(index < porras.length, "PorraFactory: index out of bounds");
61        return address(porras[index]);
62    }

64    function getPorrasByCreator(address creator) external view returns (address[] memory) {
65        return porrasByCreator[creator];
66    }

68    function setOracle(address _oracle) external onlyOwner {
69        require(_oracle != address(0), "PorraFactory: zero oracle");
70        oracle = _oracle;
71    }
72}
