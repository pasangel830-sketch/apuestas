MockOracle.sol

Oráculo simulado para desarrollo: el resultado se establece a mano (0 = gana, 1 = empate, 2 = pierde).

Código completo del contrato

 1// SPDX-License-Identifier: MIT
 2pragma solidity ^0.8.19;

 4import "./interfaces/IPorraOracle.sol";
 5import "./utils/Ownable.sol";

 7/// @title Oráculo simulado para desarrollo y tests
 8/// @notice Permite establecer el resultado manualmente (0=Atleti gana, 1=Empate, 2=Atleti pierde)
 9contract MockOracle is IPorraOracle, Ownable {
10    struct Result {
11        uint8 value;
12        bool resolved;
13    }
14    mapping(bytes32 => Result) private _results;

16    event ResultRequested(bytes32 indexed matchId);
17    event ResultSet(bytes32 indexed matchId, uint8 result);

19    constructor() Ownable(msg.sender) {}

21    function requestMatchResult(bytes32 matchId) external override {
22        emit ResultRequested(matchId);
23    }

25    /// @notice Solo el owner puede establecer el resultado (para simular respuesta del oráculo)
26    function setResult(bytes32 matchId, uint8 result) external onlyOwner {
27        require(result <= 2, "MockOracle: invalid result (0, 1 or 2)");
28        _results[matchId] = Result({ value: result, resolved: true });
29        emit ResultSet(matchId, result);
30    }

32    function getResult(bytes32 matchId) external view override returns (uint8 result, bool resolved) {
33        Result memory r = _results[matchId];
34        return (r.value, r.resolved);
35    }
36}
