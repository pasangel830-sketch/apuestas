WhitelistManager.sol

Gestiona la lista de participantes autorizados para una porra: entre 2 y 4 jugadores; solo ellos pueden apostar.

Código completo del contrato

 1// SPDX-License-Identifier: MIT
 2pragma solidity ^0.8.19;

 4import "./utils/Ownable.sol";

 6/// @title Gestión de participantes autorizados para una porra
 7/// @notice Entre 2 y 4 jugadores; solo direcciones autorizadas pueden participar
 8contract WhitelistManager is Ownable {
 9    address[] private _participants;
10    mapping(address => bool) private _isWhitelisted;
11    uint256 public constant MIN_PARTICIPANTS = 2;
12    uint256 public constant MAX_PARTICIPANTS = 4;

14    event ParticipantAdded(address indexed participant);
15    event ParticipantRemoved(address indexed participant);

17    constructor(address[] memory participants) Ownable(msg.sender) {
18        require(
19            participants.length >= MIN_PARTICIPANTS && participants.length <= MAX_PARTICIPANTS,
20            "WhitelistManager: invalid number of participants"
21        );
22        for (uint256 i = 0; i < participants.length; i++) {
23            address p = participants[i];
24            require(p != address(0), "WhitelistManager: zero address");
25            require(!_isWhitelisted[p], "WhitelistManager: duplicate participant");
26            _isWhitelisted[p] = true;
27            _participants.push(p);
28            emit ParticipantAdded(p);
29        }
30    }

32    function addParticipant(address participant) external onlyOwner {
33        require(participant != address(0), "WhitelistManager: zero address");
34        require(!_isWhitelisted[participant], "WhitelistManager: already whitelisted");
35        require(_participants.length < MAX_PARTICIPANTS, "WhitelistManager: max participants reached");
36        _isWhitelisted[participant] = true;
37        _participants.push(participant);
38        emit ParticipantAdded(participant);
39    }

41    function removeParticipant(address participant) external onlyOwner {
42        require(_isWhitelisted[participant], "WhitelistManager: not whitelisted");
43        _isWhitelisted[participant] = false;
44        for (uint256 i = 0; i < _participants.length; i++) {
45            if (_participants[i] == participant) {
46                _participants[i] = _participants[_participants.length - 1];
47                _participants.pop();
48                break;
49            }
50        }
51        emit ParticipantRemoved(participant);
52    }

54    function isWhitelisted(address account) external view returns (bool) {
55        return _isWhitelisted[account];
56    }

58    function getParticipants() external view returns (address[] memory) {
59        return _participants;
60    }

62    function participantCount() external view returns (uint256) {
63        return _participants.length;
64    }
65}
