PorraGame.sol

Una porra para un partido: 0 = gana, 1 = empate, 2 = pierde. Fases: Betting → Resolving → Claiming → Finished.

 2pragma solidity ^0.8.19;

 4import "./interfaces/IPorraOracle.sol";
 5import "./WhitelistManager.sol";
 6import "./utils/ReentrancyGuard.sol";

10/// @title Porra para un partido concreto
11/// @notice 0 = Atlético gana, 1 = Empate, 2 = Atlético pierde
12contract PorraGame is ReentrancyGuard, Pausable, Ownable {
13    enum GameState {
14        Betting,
15        Resolving,
16        Claiming,
17        Finished
18    }

20    uint256 public stake;
21    uint256 public constant MIN_PLAYERS_TO_RESOLVE = 2;

23    WhitelistManager public whitelistManager;
24    IPorraOracle public oracle;
25    bytes32 public matchId;
26    uint256 public bettingDeadline;
27    uint256 public matchEndTime;

29    GameState public gameState;
30    uint8 public finalResult;
31    uint256 public totalPot;
32    mapping(address => uint8) public bets;
33    mapping(address => bool) public hasPlacedBet;
34    mapping(address => bool) public hasClaimed;
35    address[] private _playersWhoBet;

37    event BetPlaced(address indexed player, uint8 prediction);
38    event ResolutionRequested(bytes32 indexed matchId);
39    event GameResolved(uint8 result);
40    event RewardClaimed(address indexed player, uint256 amount);
41    event ManualResolutionSet(uint8 result);

43    constructor(
44        address _whitelistManager,
45        address _oracle,
46        bytes32 _matchId,
47        uint256 _bettingDeadline,
48        uint256 _matchEndTime,
49        uint256 _stake
50    ) Ownable(msg.sender) {
51        require(_stake > 0, "PorraGame: stake must be > 0");
52        whitelistManager = WhitelistManager(payable(_whitelistManager));
53        oracle = IPorraOracle(_oracle);
54        matchId = _matchId;
55        bettingDeadline = _bettingDeadline;
56        matchEndTime = _matchEndTime;
57        stake = _stake;
58        gameState = GameState.Betting;
59    }

61    function placeBet(uint8 prediction) external payable nonReentrant whenNotPaused {
62        require(gameState == GameState.Betting, "PorraGame: not in betting phase");
63        require(block.timestamp <= bettingDeadline, "PorraGame: betting deadline passed");
64        require(msg.value == stake, "PorraGame: stake amount mismatch");
65        require(prediction <= 2, "PorraGame: prediction must be 0, 1 or 2");
66        require(whitelistManager.isWhitelisted(msg.sender), "PorraGame: not whitelisted");
67        require(!hasPlacedBet[msg.sender], "PorraGame: already bet");

69        hasPlacedBet[msg.sender] = true;
70        bets[msg.sender] = prediction;
71        _playersWhoBet.push(msg.sender);
72        totalPot += msg.value;

74        emit BetPlaced(msg.sender, prediction);
75    }

77    function _hasBet(address player) internal view returns (bool) {
78        for (uint256 i = 0; i < _playersWhoBet.length; i++) {
79            if (_playersWhoBet[i] == player) return true;
80        }
81        return false;
82    }

85    function startResolution() external nonReentrant whenNotPaused {
86        require(gameState == GameState.Betting, "PorraGame: not in betting phase");
87        require(block.timestamp >= matchEndTime, "PorraGame: match not ended");
88        require(_playersWhoBet.length >= MIN_PLAYERS_TO_RESOLVE, "PorraGame: min 2 players to resolve");

90        gameState = GameState.Resolving;
91        oracle.requestMatchResult(matchId);
92        emit ResolutionRequested(matchId);
93    }

96    function resolveWithOracle() external nonReentrant whenNotPaused {
97        require(gameState == GameState.Resolving, "PorraGame: not in resolving phase");
98        (uint8 result, bool resolved) = oracle.getResult(matchId);
99        require(resolved, "PorraGame: oracle has not responded");

101        finalResult = result;
102        gameState = GameState.Claiming;
103        emit GameResolved(result);
104    }

107    function setManualResult(uint8 result) external onlyOwner whenNotPaused {
108        require(gameState == GameState.Resolving, "PorraGame: not in resolving phase");
109        require(result <= 2, "PorraGame: result must be 0, 1 or 2");

111        finalResult = result;
112        gameState = GameState.Claiming;
113        emit ManualResolutionSet(result);
114    }

117    function claimReward() external nonReentrant whenNotPaused {
118        require(gameState == GameState.Claiming || gameState == GameState.Finished, "PorraGame: not in claiming phase");
119        require(hasPlacedBet[msg.sender], "PorraGame: did not participate");
120        require(!hasClaimed[msg.sender], "PorraGame: already claimed");

122        hasClaimed[msg.sender] = true;

124        uint256 amount = _claimableAmount(msg.sender);
125        require(amount > 0, "PorraGame: nothing to claim");

127        if (gameState == GameState.Claiming) {
128            gameState = GameState.Finished;
129        }

131        (bool sent, ) = msg.sender.call{ value: amount }("");
132        require(sent, "PorraGame: transfer failed");
133        emit RewardClaimed(msg.sender, amount);
134    }

136    function _claimableAmount(address player) internal view returns (uint256) {
137        uint256 winners = 0;
138        for (uint256 i = 0; i < _playersWhoBet.length; i++) {
139            if (bets[_playersWhoBet[i]] == finalResult) winners++;
140        }
141        if (winners == 0) {
142            return hasPlacedBet[player] ? stake : 0;
143        }
144        uint8 prediction = bets[player];
145        if (prediction != finalResult) return 0;
146        return totalPot / winners;
147    }

150    function getClaimableAmount(address player) external view returns (uint256) {
151        if (gameState != GameState.Claiming && gameState != GameState.Finished) return 0;
152        if (hasClaimed[player]) return 0;
153        if (!hasPlacedBet[player]) return 0;
154        return _claimableAmount(player);
155    }

158    function getPlayersWhoBet() external view returns (address[] memory) {
159        return _playersWhoBet;
160    }

163    function getBets() external view returns (uint8[] memory) {
164        uint8[] memory b = new uint8[](_playersWhoBet.length);
165        for (uint256 i = 0; i < _playersWhoBet.length; i++) {
166            b[i] = bets[_playersWhoBet[i]];
167        }
168        return b;
169    }

172    function getContractState() external view returns (
173        GameState state,
174        uint256 pot,
175        uint256 deadline,
176        uint8 result,
177        bool resolved
178    ) {
179        bool isResolved = (gameState == GameState.Claiming || gameState == GameState.Finished);
180        return (gameState, totalPot, bettingDeadline, finalResult, isResolved);
181    }

183    function pause() external onlyOwner {
184        _pause();
185    }

187    function unpause() external onlyOwner {
188        _unpause();
189    }
190}
